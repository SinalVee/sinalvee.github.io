// hdoj_2003 求绝对值
// 0MS	244K	181 B	GCC

#include <stdio.h>
int main(void)
{
	double x;
	while(scanf("%lf", &x) != EOF)
	{
		if(x > 0)
			printf("%.2lfn", x);
		else
			printf("%.2lfn", -x);
	}
	return 0;
}
