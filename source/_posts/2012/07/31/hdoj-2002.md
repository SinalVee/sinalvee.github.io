// hdoj_2002 计算球体积
// 15MS	244K	188 B	GCC

#include <stdio.h>
#define PI 3.1415927
int main(void)
{
	double r, s;
	while(scanf("%lf", &r) != EOF)
	{
		s = 4.0/3.0 * PI * r * r * r;
		printf("%.3lfn", s);
	}
	return 0;
}
