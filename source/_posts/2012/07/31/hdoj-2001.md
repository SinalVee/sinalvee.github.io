```c
// hdoj_2001 计算两点间的距离
//0MS	244K	245 B	GCC

#include <stdio.h>
#include <math.h>
int main(void)
{
	double x1, x2, y1, y2, s;
	while(scanf("%lf %lf %lf %lf", &x1, &y1, &x2, &y2) != EOF)
	{
		s = sqrt((x2-x1) * (x2-x1) + (y2-y1) * (y2-y1));
		printf("%.2lfn", s);
	}
	return 0;
}
```
