```c
// hdoj_1096 A+B for Input-Output Practice (VIII)

#include <stdio.h>
int main(void)
{
	int n, m, a, sum, i, j;
	scanf("%d", &n);
    for(i = 0; i < n; i ++)
    {
        scanf("%d", &m);
        sum = 0;
        for(j = 0; j < m; j ++)
        {
            scanf("%d", &a);
            sum = sum + a;
        }
        printf("%dn", sum);
        if(i != n - 1) // 这里是这道题的重点
            printf("n");    
    }
    return 0;
}
```

每两个输出之间有空行而最后一个输出之后没有空行。
