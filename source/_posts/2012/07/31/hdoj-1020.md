// hdoj_1020 Encoding

#include <stdio.h>
#include <string.h>
int main(void)
{
	int i, j, k, n, len, c[26] = {0};
	char a[10000], b[26];
	scanf("%d", &n);
	getchar();        //接收上面的回车，以免被后面gets()接收
	for(i = 0; i < n; i ++)
	{
	  for(j = 0; j < 26; j ++)
      c[j] = 0;
		gets(a);
		len = strlen(a);
		b[0] = a[0];
		for(j = 0, k = 0; j < len; j ++)
		{
			if(b[k] == a[j])
			{
				c[k]++;
			}
			else
			{
				b[k+1] = a[j];
				k++;
				c[k]++;
			}
		}
		for(j = 0; j <= k; j ++)
		{
			if(c[j] != 1)
				printf("%d%c", c[j], b[j]);
			else
				printf("%c", b[j]);
		}
		printf("n");
	}
	return 0;
}
