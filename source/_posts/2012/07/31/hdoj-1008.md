// hdoj_1008 Elevator 电梯

#include <stdio.h>
int main()
{
	int n, i, time, a[100];
	while(scanf("%d", &n) && n != 0)
	{
	    a[0] = 0;
		time = 0;
		for(i = 1; i <= n; i ++)
			scanf("%d", &a[i]);
		for(i = 1; i <= n; i ++)
			if((a[i] - a[i-1]) > 0)
				time = time + (a[i] - a[i-1]) * 6 + 5;
			else
				time = time + (a[i-1] - a[i]) * 4 + 5;
		printf("%dn", time);
	}
	return 0;
}
