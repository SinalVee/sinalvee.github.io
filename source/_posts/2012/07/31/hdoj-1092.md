```c
// hdoj_1092 A+B for Input-Output Practice (IV)

#include <stdio.h>
int main(void)
{
	int a, n, i, sum;
	while(scanf("%d", &n) && n != 0) // while(scanf("%d", &n) != 0)是错误的用法
	{
	    sum = 0;
		for(i = 0; i < n; i ++)
		{
			scanf("%d", &a);
			sum = sum + a;
		}
		printf("%dn", sum);
	}

	return 0;
}
```
