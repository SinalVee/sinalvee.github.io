// hdoj_2036 改（和谐）革春风吹满地
// 0MS	236K	393 B	GCC

#include <stdio.h>
int main(void)
{
	int n, i;
	__int64 x[101], y[101];
	while(scanf("%d", &n) && n != 0)
	{
		for(i = 0; i < n; i ++)
			scanf("%I64d%I64d", &x[i], &y[i]);
		x[n] = x[0];
		y[n] = y[0];
		double area = 0;
		for(i = 0; i < n; i ++)
			area += 0.5 * (x[i] - x[i+1]) * (y[i] + y[i+1]);
		if(area < 0)
			area *= -1;
		printf("%.1lfn", area);
	}
	return 0;
}
