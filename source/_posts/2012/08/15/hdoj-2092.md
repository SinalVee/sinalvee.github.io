// hdoj_2092 整数解
// 234MS	220K	279 B	GCC

#include <stdio.h>
int main(void)
{
	int n, m, i, c;
	while(scanf("%d%d", &n, &m) && n || m)
	{
		c = 0;
		for(i = -9999; i < 10000; i ++)
		{
			if(i*(n-i) == m)
			{
				c = 1;
			}
		}
		if(c)
			printf("Yesn");
		else
			printf("Non");
	}
	return 0;
}
