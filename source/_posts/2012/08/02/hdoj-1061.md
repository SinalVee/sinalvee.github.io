```c
// hdoj_1061 Rightmost Digit 右值
// 0MS	228K	253 B	GCC

#include <stdio.h>
int main(vodi)
{
	int i, ncase, m, n, t;
	scanf("%d", &ncase);
	while(ncase--)
	{
		scanf("%ld", &n);
		t = n % 10;
		m = 1;
		for(i = 0; i <= (n-1)%4; i ++) //此处可换为 i < n%4 + 4;可解决n%4 ==0的情况
		{
			m *= t;;
		}
		printf("%dn", m%10);
	}
	return 0;
}
```
