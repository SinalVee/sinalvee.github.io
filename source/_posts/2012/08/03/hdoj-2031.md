// hdoj_2031 进制转换
// 0MS	232K	407 B	GCC

#include <stdio.h>
int main(void)
{
	int m, n, p, q, a[100];
	while(scanf("%d%d", &m, &n) != EOF)
	{
		p = 0;
		q = 0;
		if(m < 0)
		{
			m = -1*m;
			p = 1;
		}
		do
		{
			a[q++] = m%n;
			m = m/n;
		}
		while(m);
		if(p == 1)
			printf("-");
		while(q--)
		{
			if(a[q] > 9)
				printf("%c", 'A'+a[q]-10);
			else
				printf("%d", a[q]);
		}
		printf("n");
	}
	return 0;
}
