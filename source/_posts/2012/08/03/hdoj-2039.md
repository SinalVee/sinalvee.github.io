// hdoj_2039 三角形
// 0MS	224K	224 B	GCC

#include <stdio.h>
int main(void)
{
  int n;
	float a, b, c;   //注意审题，题目要求是三个正数，而非整数
	scanf("%f", &n);
	while(n--)
	{
		scanf("%f%f%f", &a, &b, &c);
		if(a+b>c && a+c>b && b+c>a)
			printf("YESn");
		else
			printf("NOn");
	}
	return 0;
}
