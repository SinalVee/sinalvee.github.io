// hdoj_2026 首字母变大写
// 0MS	208K	307 B	GCC

#include <stdio.h>
#include <string.h>
int main(void)
{
	int i, len;
	char a[100];
	while(gets(a))
	{
		len = strlen(a);
		for(i = 1; i < len; i ++)
		{
			if(a[i-1] == ' ' && a[i] != ' ')
				a[i] = a[i] - 32;
		}
		if(a[0] != ' ')
			a[0] = a[0] - 32;
        puts(a);
	}
	return 0;
}
