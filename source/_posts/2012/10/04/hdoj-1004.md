// hdoj_1004 Let the Balloon Rise
// 0MS 216K 983 B GCC

#include <stdio.h>
#include <string.h>
#define MAXN 1000+10
char a[MAXN][15];
int b[MAXN];
int main(void)
{
    int n;
    char temp[15];
    while(scanf("%d", &n) && n)
    {
        memset(b, 0, sizeof(b));
        int i = 0, j, m = 0, ok, max, maxx;
        for(i = 0; i < n; i++)
        {
            scanf("%s", temp);
            j = 0;
            ok = 0;
            for(j = 0; j < m; j++)
                if(strcmp(a[j], temp) == 0)
                {
                    b[j]++;
                    ok = 1;
                    break;
                }
                if(ok == 0)
                {
                    strcpy(a[m], temp);
                    b[m]++;
                    m++;
                }
        }
        max = b[0];
        maxx = 0;
        for(i = 0; i <= m; i++)
            if(b[i] > max)
            {
                max = b[i];
                maxx = i;
            }
        printf("%sn", a[maxx]);
    }
    return 0;
}
